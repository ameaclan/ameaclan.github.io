    <script>
        // JavaScript code to update visitor count
        document.addEventListener('DOMContentLoaded', function() {
            var countSpan = document.getElementById('visitorCount');
            if (localStorage.getItem('visits')) {
                countSpan.textContent = localStorage.getItem('visits');
            } else {
                countSpan.textContent = '0';
            }
            var visits = parseInt(localStorage.getItem('visits')) || 0;
            visits++;
            localStorage.setItem('visits', visits);
            countSpan.textContent = visits;
        });
    </script>
